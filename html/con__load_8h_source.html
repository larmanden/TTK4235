6// Load values from a config file
7//
8// Key-value pairs in the config file are assumed to be of the form:
9// "--key value"
10// Lines not starting in "--" are ignored.
11// Keys are *not* case-sensitive
12// Enum values are *not* case-sensitive
13//
14// keyuments:
15// file: Name of the file to load.
16// cases: One or more instance of `con_val()` or `con_enum()`
17// The cases must *not* be separated by commas.
18//
19// Example:
20// /* Content of "config.con":
21// ```
22// --integer 5
23// --greeting hello
24// --enumeration En2
25// ```
26// */
27//
28// typedef enum { En1, En2, En3 } En;
29// int i;
30// char s[16];
31// En en;
32//
33// con_load("config.con",
34// con_val("integer", &i, "%d")
35// con_val("greeting", s, "%[^\n]")
36// con_enum("enumeration", &en,
37// con_match(En1)
38// con_match(En2)
39// con_match(En3)
40// )
41// )
42// printf("%s, %d, %d\n", s, i, en); // Should print "hello, 5, 1"
43//
44#define con_load(file, cases) \
45{ \
46 FILE* _f = fopen(file, "r"); \
47 if(_f){ \
48 char _line[128] = {0}; \
49 while(fgets(_line, 128, _f)){ \
50 if(!strncmp(_line, "--", 2)){ \
51 char _key[64]; \
52 char _val[64]; \
53 sscanf(_line, "--%s %s", _key, _val); \
54 cases \
55 } \
56 } \
57 } else { \
58 printf("Unable to open config file %s\n", file); \
59 } \
60}
