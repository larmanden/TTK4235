81void fsm_moving(elevator* el){
82
83 if(elevio_stopButton()){
84 el->state = EMERGENCY_STOP;
85 elevator_update_dir(el, DIRN_STOP);
86 return;
87 }
88
89 printf("MOVING \n");
90 queue_print(el->queue);
91 queue_update(el->queue);
92 elevator_btnlights_update(el->queue);
93 elevator_update_current_floor(el);
94
95 //Deciding wether to stop or not
96 int tempfloor = elevio_floorSensor();
97 if(tempfloor != -1){
98 elevio_floorIndicator(tempfloor);
99 switch (el->current_motor_dir){
100 case DIRN_DOWN:
101 if(el->queue[BUTTON_HALL_DOWN][tempfloor] == 1 || el->queue[BUTTON_CAB][tempfloor] == 1){
102 elevator_update_dir(el, DIRN_STOP);
103 el->state = DOOR_OPEN;
104 }
105 break;
106 case DIRN_UP:
107 if(el->queue[BUTTON_HALL_UP][tempfloor] == 1 || el->queue[BUTTON_CAB][tempfloor] == 1){
108 elevator_update_dir(el, DIRN_STOP);
109 el->state = DOOR_OPEN;
110 }
111 break;
112 case DIRN_STOP:
113 if(el->queue[BUTTON_HALL_DOWN][tempfloor] == 1 || el->queue[BUTTON_HALL_UP][tempfloor] == 1 || el->queue[BUTTON_CAB][tempfloor] == 1){
114 elevator_update_dir(el, DIRN_STOP);
115 el->state = DOOR_OPEN;
116 }
117 break;
118 default:
119 break;
120 }
121
122 if(tempfloor == elevator_look_ahead(el)){
123 elevator_update_dir(el, DIRN_STOP);
124 el->state = DOOR_OPEN;
125 }
126 }
127 //Setting motordirection based on target
128 switch (el->current_motor_dir){
129 case DIRN_DOWN:
130 if(elevator_order_below(el)){
131 elevator_update_dir(el, DIRN_DOWN);
132 }
133 else if (elevator_order_above(el)){
134 elevator_update_dir(el, DIRN_UP);
135 }
136 break;
137 case DIRN_STOP:
138 if(elevator_order_above(el)){
139 elevator_update_dir(el, DIRN_UP);
140 }
141 else if (elevator_order_below(el)){
142 elevator_update_dir(el, DIRN_DOWN);
143 }
144 else if(elevio_floorSensor() == -1){
145 MotorDirection newdir = elevator_move_after_emergency(el);
146 elevator_update_dir(el, newdir);
147 }
148 break;
149 case DIRN_UP:
150 if(elevator_order_above(el)){
151 elevator_update_dir(el, DIRN_UP);
152 }
153 else if (elevator_order_below(el)){
154 elevator_update_dir(el, DIRN_DOWN);
155 }
156 break;
157 }
